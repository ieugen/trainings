NamedParameterJdbcTemplate option

Just like JdbcTemplate but with named parmeters


    template = new JdbcTemplate(dataSource);
    String sql = "select count(*) from person where name = :name";
    SqlParameterSource namedParameters = new MapSqlParameterSource("name", "Gabi");
    int rowCount = template.queryForObject(sql, namedParameters, Integer.class);
